001    package com.googlecode.catchexception;
002    
003    import java.io.Serializable;
004    
005    /**
006     * Thrown if a method has not thrown an exception of the expected type.
007     * 
008     * @author rwoo
009     * @since 16.09.2011
010     */
011    public class ExceptionNotThrownAssertionError extends AssertionError {
012    
013        /**
014         * See {@link Serializable}.
015         */
016        private static final long serialVersionUID = 7044423604241785057L;
017    
018        /**
019         * Use this constructor if neither an exception of the expected type nor
020         * another exception is thrown.
021         * 
022         * @param <E>
023         *            the type of the exception that is not thrown.
024         * @param clazz
025         *            the type of the exception that is not thrown.
026         */
027        public <E extends Exception> ExceptionNotThrownAssertionError(Class<E> clazz) {
028            super(clazz == Exception.class ? "Exception expected but not thrown"
029                    : "Neither an exception of type " + clazz.getName()
030                            + " nor another exception was thrown");
031        }
032    
033        /**
034         * Use this constructor if an exception of another than the expected type is
035         * thrown.
036         * 
037         * @param <E>
038         *            the type of the exception that is not thrown.
039         * @param clazz
040         *            the type of the exception that is not thrown.
041         * @param e
042         *            the exception that has been thrown instead of the expected
043         *            one.
044         */
045        public <E extends Exception> ExceptionNotThrownAssertionError(
046                Class<E> clazz, Exception e) {
047            super("Exception of type " + clazz.getName()
048                    + " expected but was not thrown. "
049                    + "Instead an exception of type " + e.getClass()
050                    + " with message '" + e.getMessage() + "' was thrown.");
051        }
052    }
