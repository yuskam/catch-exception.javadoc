001    package com.googlecode.catchexception.internal;
002    
003    import org.mockito.cglib.proxy.MethodInterceptor;
004    
005    /**
006     * Creates proxies.
007     * 
008     * @author rwoo
009     * 
010     */
011    interface ProxyFactory {
012    
013        /**
014         * Create a proxy.
015         * 
016         * @param <T>
017         *            The type parameter makes some casts redundant.
018         * @param targetClass
019         *            the class the factory shall create a proxy for
020         * @param interceptor
021         *            the method interceptor that shall be applied to all method
022         *            calls.
023         * @return Returns the created proxy.
024         */
025        public <T> T createProxy(Class<?> targetClass, MethodInterceptor interceptor);
026    
027    }
