001    package com.googlecode.catchexception.internal;
002    
003    import org.mockito.cglib.proxy.MethodInterceptor;
004    import org.mockito.exceptions.base.MockitoException;
005    import org.mockito.internal.creation.jmock.ClassImposterizer;
006    
007    /**
008     * This {@link ProxyFactory} uses Mockito's jmock package to create proxies that
009     * subclass from the target's class.
010     * 
011     * @author rwoo
012     */
013    public class SubclassProxyFactory implements ProxyFactory {
014    
015        /**
016         * That proxy factory is used if this factory cannot be used.
017         */
018        private ProxyFactory fallbackProxyFactory = new InterfaceOnlyProxyFactory();
019    
020        /*
021         * (non-Javadoc)
022         * 
023         * @see
024         * com.googlecode.catchexception.internal.ProxyFactory#createProxy(java.
025         * lang.Object, java.lang.Class, boolean)
026         */
027        @SuppressWarnings("unchecked")
028        public <T> T createProxy(Class<?> targetClass, MethodInterceptor interceptor) {
029    
030            // can we subclass the class of the target?
031            if (!ClassImposterizer.INSTANCE.canImposterise(targetClass)) {
032    
033                // delegate
034                return fallbackProxyFactory.<T> createProxy(targetClass,
035                        interceptor);
036            }
037    
038            // create proxy
039            T proxy;
040            try {
041    
042                proxy = (T) ClassImposterizer.INSTANCE.imposterise(interceptor,
043                        targetClass, SubclassProxy.class);
044    
045            } catch (MockitoException e) {
046    
047                // delegate
048                return fallbackProxyFactory.<T> createProxy(targetClass,
049                        interceptor);
050            }
051    
052            return proxy;
053        }
054    }
