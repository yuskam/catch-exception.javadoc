001    package com.googlecode.catchexception.internal;
002    
003    import java.lang.reflect.Method;
004    
005    import org.mockito.cglib.proxy.MethodInterceptor;
006    import org.mockito.cglib.proxy.MethodProxy;
007    
008    /**
009     * This {@link AbstractExceptionProcessingInvocationHandler} implements
010     * {@link MethodInterceptor} for Mockito's cglib variant.
011     * 
012     * @author rwoo
013     * @param <E>
014     *            The type of the exception that shall be caught and (optionally)
015     *            verified
016     */
017    public class ExceptionProcessingInterceptor<E extends Exception>
018            extends AbstractExceptionProcessingInvocationHandler<E> implements
019            MethodInterceptor {
020    
021        @SuppressWarnings("javadoc")
022        public ExceptionProcessingInterceptor(Object target,
023                Class<E> clazz, boolean assertException) {
024            super(target, clazz, assertException);
025        }
026    
027        /*
028         * (non-Javadoc)
029         * 
030         * @see
031         * org.mockito.cglib.proxy.MethodInterceptor#intercept(java.lang.Object,
032         * java.lang.reflect.Method, java.lang.Object[],
033         * org.mockito.cglib.proxy.MethodProxy)
034         */
035        public Object intercept(Object obj, Method method, Object[] args,
036                MethodProxy proxy) throws Throwable {
037    
038            beforeInvocation();
039    
040            try {
041    
042                Object retval = proxy.invoke(target, args);
043    
044                return afterInvocation(retval);
045    
046            } catch (Exception e) {
047    
048                return afterInvocationThrowsException(e, method);
049            }
050        }
051    }
