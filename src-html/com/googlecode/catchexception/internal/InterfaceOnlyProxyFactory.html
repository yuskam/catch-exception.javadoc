001    package com.googlecode.catchexception.internal;
002    
003    import java.util.HashSet;
004    import java.util.Set;
005    
006    import org.mockito.cglib.proxy.Enhancer;
007    import org.mockito.cglib.proxy.MethodInterceptor;
008    
009    /**
010     * This {@link ProxyFactory} create proxies that implements all interfaces of
011     * the underlying object including the marker interface
012     * {@link InterfaceOnlyProxy}. But in contrast to the proxies created by
013     * {@link SubclassProxyFactory} such a proxy does not subclass the class of the
014     * underlying object.
015     * 
016     * @author rwoo
017     */
018    public class InterfaceOnlyProxyFactory implements ProxyFactory {
019    
020        /*
021         * (non-Javadoc)
022         * 
023         * @see
024         * com.googlecode.catchexception.internal.ProxyFactory#createProxy(java.
025         * lang.Class, org.mockito.cglib.proxy.MethodInterceptor)
026         */
027        @SuppressWarnings("unchecked")
028        public <T> T createProxy(Class<?> targetClass, MethodInterceptor interceptor) {
029    
030            // get all the interfaces (is there an easier way?)
031            Set<Class<?>> interfaces = new HashSet<Class<?>>();
032            Class<?> clazz = targetClass;
033            while (true) {
034                for (Class<?> interfaze : clazz.getInterfaces()) {
035                    interfaces.add(interfaze);
036                }
037                if (clazz.getSuperclass() == null) {
038                    break;
039                }
040                clazz = clazz.getSuperclass();
041            }
042            interfaces.add(InterfaceOnlyProxy.class);
043    
044            return (T) Enhancer.create(Object.class,
045                    interfaces.toArray(new Class<?>[interfaces.size()]),
046                    interceptor);
047        }
048    }
