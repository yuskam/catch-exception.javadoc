001    package com.googlecode.catchexception.internal;
002    
003    import java.lang.ref.WeakReference;
004    
005    /**
006     * Holds a caught exception {@link ThreadLocal per Thread}.
007     * 
008     * @author rwoo
009     * 
010     */
011    public class ExceptionHolder {
012    
013        /**
014         * The container for the most recently caught exception.
015         * <p>
016         * There is no need to use {@link WeakReference weak references} here as all
017         * the code is for testing so that we don't have to care about memory leaks.
018         */
019        private static final ThreadLocal<Exception> caughtException = new ThreadLocal<Exception>();
020    
021        /**
022         * Saves the given exception in {@link #caughtException}.
023         * 
024         * @param <E>
025         *            the type of the caught exception
026         * @param caughtException
027         *            the caught exception
028         */
029        public static <E extends Exception> void set(E caughtException) {
030            ExceptionHolder.caughtException.set(caughtException);
031        }
032    
033        /**
034         * @param <E>
035         *            the type of the caught exception
036         * @return Returns the caught exception. Returns null if there is no
037         *         exception caught.
038         */
039        @SuppressWarnings("unchecked")
040        public static <E extends Exception> E get() {
041            return (E) caughtException.get();
042        }
043    
044    }
